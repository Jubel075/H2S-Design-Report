\subsection*{Flow Rate Conversions}
\label{subsec:flow_conversions}

\textbf{Standard conditions:} 60°F (15.56°C), 14.696 psia (101.325 kPa)

\textbf{Total molar flow rate:}
\begin{equation}
  \dot{n}_{\text{total}} = \frac{660{,}000 \text{ SCF/day}}{379.5 \text{ SCF/lbmol}} = 1{,}739 \text{ lbmol/day} = 72.5 \text{ lbmol/h} = 32.9 \text{ kmol/h}
\end{equation}

\textbf{Mass flow rate:}
\begin{equation}
  \dot{m}_{\text{gas}} = 32.9 \text{ kmol/h} \times 17.13 \text{ kg/kmol} = 564 \text{ kg/h} = 13.5 \text{ ton/day}
\end{equation}

\subsection*{\ce{H2S} Absorption Material Balance}
\label{subsec:h2s_balance}

\subsubsection*{Stoichiometry}

The primary absorption reaction is:
\begin{equation}
  \ce{H2S + NaOH -> NaHS + H2O}
  \label{eq:h2s_stoich}
\end{equation}

At high pH (>12), the secondary reaction may occur:
\begin{equation}
  \ce{NaHS + NaOH -> Na2S + H2O}
  \label{eq:h2s_secondary}
\end{equation}

For design purposes, 1:1 stoichiometry (Eq.~\ref{eq:h2s_stoich}) is assumed as the predominant reaction.

\subsubsection*{Molar Flow Calculations}

\textbf{\ce{H2S} in feed:}
\begin{equation}
  \dot{n}_{\ce{H2S},\text{in}} = 0.0018 \times 32.9 \text{ kmol/h} = 0.0592 \text{ kmol/h} = 1.42 \text{ kmol/day}
\end{equation}

\textbf{\ce{H2S} absorbed (at 97.5\% removal):}
\begin{equation}
  \dot{n}_{\ce{H2S},\text{abs}} = 0.975 \times 0.0592 = 0.0578 \text{ kmol/h} = 1.39 \text{ kmol/day}
\end{equation}

\textbf{\ce{NaOH} consumption for \ce{H2S}:}
\begin{equation}
  \dot{m}_{\ce{NaOH},\ce{H2S}} = 0.0578 \text{ kmol/h} \times 40 \text{ kg/kmol} = 2.31 \text{ kg/h} = 55 \text{ kg/day}
\end{equation}

\textbf{Sulfide load to treatment:}
\begin{equation}
  \dot{m}_{\text{S}} = 0.0578 \text{ kmol/h} \times 32 \text{ kg/kmol} = 1.85 \text{ kg~S/h} = 44 \text{ kg~S/day}
\end{equation}

\subsection*{\ce{CO2} Co-absorption Scenarios}
\label{subsec:co2_scenarios}

The \ce{CO2} absorption depends on contactor type and operating pH. The absorption stoichiometry is:
\begin{equation}
  \ce{CO2 + 2 NaOH -> Na2CO3 + H2O}
  \label{eq:co2_stoich}
\end{equation}

\textbf{\ce{CO2} in feed:}
\begin{equation}
  \dot{n}_{\ce{CO2},\text{in}} = 0.0286 \times 32.9 \text{ kmol/h} = 0.941 \text{ kmol/h} = 22.6 \text{ kmol/day}
\end{equation}

\subsection*{Key Observations}
\label{subsec:balance_observations}

\begin{enumerate}
  \item \textbf{Sulfide load is constant:} The sulfide load to biological treatment (44~kg~S/day) is independent of scrubber technology because both systems remove the same amount of \ce{H2S}.

  \item \textbf{\ce{NaOH} consumption dominates economics:} The 9:1 ratio of \ce{NaOH} consumption between packed column (70\% \ce{CO2}) and SCT (5\% \ce{CO2}) directly translates to operating cost difference.

  \item \textbf{Carbonate management differs:} The packed column produces 14$\times$ more carbonate than SCT, requiring larger blowdown volumes and more complex spent caustic composition.

  \item \textbf{Solution inventory:} At 50\% \ce{NaOH} solution, the packed column requires 880 L/day makeup versus 97 L/day for SCT, affecting storage and logistics requirements.
\end{enumerate}
